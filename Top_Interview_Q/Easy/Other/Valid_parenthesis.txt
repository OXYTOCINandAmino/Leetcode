class Solution:
    def isValid(self, s: str) -> bool:
        D = {"(":")", "[":"]","{":"}"}
        L = []
        for i in s:
            if(i in D):
                L.append(i)
            else:
                if(len(L)< 1):
                    return False
                
                top = L.pop()
                if(D[top] != i):
                    return False
        if(len(L)!=0):
            return False
        return True
